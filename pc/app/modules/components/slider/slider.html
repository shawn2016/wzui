滑块

通过拖动滑块在一个固定区间内进行选择。 


<input type="text" id="range" value="" name="range" />



 $("#range").ionRangeSlider({
            hide_min_max: false,
            keyboard: false,
            min: 0,
            max: 5000,
            from: 1250,
            to: 3750,
            type: 'double',
            step: 1,
            prefix: "$",//前缀
            grid: true,//刻度
            disable:true,//禁用
            postfix: "元",//后缀
            hide_from_to:false//隐藏显示框
        });
