alert 弹层

共15种不同的状态只需复制下面代码就可实现你想要的效果 

        
        <!-- 只有内容 -->
        <div id="alert1" class="rob-alert" style="display:none">
          <div class="rob-alert-content">
          这是一段内容~ 
          啊啊啊啊啊啊啊啊啊啊啊啊啊， 
          啊啊啊啊啊啊啊啊啊啊
          </div>
      </div>
  


        <!-- 内容+内标题+右上关闭按钮 -->

          <span aria-hidden="true">×</span>
          </button>
            <div class="rob-alert-content">
            <div class="rob-alert-content-title">这是一个标题</div>
            这是一段内容~ 
            </div>
         </div>
